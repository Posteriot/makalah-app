  Implementasi Step-by-Step (Low Pain Path)

  Phase 1: Backend API Route

  File: src/app/api/refrasa/route.ts

  // Pseudocode structure
  export async function POST(request: Request) {
    const { text } = await request.json()

    // Step 1: Validate original
    const validation = BahasaStyle.validate(text)

    // Step 2: If already good, return early
    if (validation.isValid && validation.score >= 80) {
      return Response.json({
        needsRefrasa: false,
        score: validation.score
      })
    }

    // Step 3: Generate refrasa dengan AI + specific feedback
    const refrasaPrompt = buildRefrasaPrompt(text, validation.issues)
    const refrasedText = await generateWithAI(refrasaPrompt)

    // Step 4: Validate result
    const newValidation = BahasaStyle.validate(refrasedText)

    return Response.json({
      needsRefrasa: true,
      original: { text, score: validation.score, issues: validation.issues },
      refrasad: { text: refrasedText, score: newValidation.score, issues: newValidation.issues }
    })
  }

  Indonesian sentence splitting sebenarnya straightforward:
  // Simple Indonesian sentence tokenizer
  function splitSentences(text: string): string[] {
    // Split by . ! ? tapi handle abbreviations common di Indonesian
    const abbreviations = ['dr.', 'prof.', 'dkk.', 'dsb.', 'dll.', 'sdr.', 'no.', 'hal.']

    // Protect abbreviations
    let protected = text
    abbreviations.forEach((abbr, i) => {
      protected = protected.replace(new RegExp(abbr, 'gi'), `__ABBR${i}__`)
    })

    // Split
    const sentences = protected.split(/(?<=[.!?])\s+/)

    // Restore abbreviations
    return sentences.map(s => {
      abbreviations.forEach((abbr, i) => {
        s = s.replace(new RegExp(`__ABBR${i}__`, 'g'), abbr)
      })
      return s.trim()
    }).filter(s => s.length > 0)
  }
